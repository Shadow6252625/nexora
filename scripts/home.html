index.ts
import { NexoraClient, Network } from '@nexus-labs/sdk';

// Initialize Client
const nexus = new NexoraClient({
  network: Network.Mainnet,
  rpc: 'https://api.mainnet-beta.solana.com'
});

// Appraise and Transmute
async function liquidateAsset(mintAddress) {
  const valuation = await nexus.oracle.getValuation(mintAddress);
  
  console.log(`Fair Value: ${valuation.price} SOL`);
  
  const tx = await nexus.liquidity.createPool({
    mint: mintAddress,
    curve: 'exponential',
    instantLiquidity: true
  });
  
  return tx.signature;
}
